53 static glitchedhttps_header* glitchedhttps_header_init(const char* type, const size_t type_length, const char* value, const size_t value_length)
54 {
55  if (type == NULL || value == NULL)
56  {
57  _glitchedhttps_log_error("Header type or value string NULL!", __func__);
58  return NULL;
59  }
60 
61  if (type_length == 0)
62  {
63  _glitchedhttps_log_error("Header type string empty!", __func__);
64  return NULL;
65  }
66 
67  glitchedhttps_header* out = malloc(sizeof(glitchedhttps_header));
68  if (out == NULL)
69  {
70  _glitchedhttps_log_error("OUT OF MEMORY!", __func__);
71  return NULL;
72  }
73 
74  out->type = malloc(sizeof(char) * type_length + 1);
75  out->value = malloc(sizeof(char) * value_length + 1);
76 
77  if (out->type == NULL || out->value == NULL)
78  {
79  _glitchedhttps_log_error("OUT OF MEMORY!", __func__);
80  return NULL;
81  }
82 
83  memcpy(out->type, type, type_length);
84  out->type[type_length] = '\0';
85 
86  if (value_length > 0)
87  {
88  memcpy(out->value, value, value_length);
89  out->value[value_length] = '\0';
90  }
91  else
92  {
93  out->value[0] = '\0';
94  }
95 
96  return out;
97 }
