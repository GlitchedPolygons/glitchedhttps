39GLITCHEDHTTPS_API int glitchedhttps_strncmpic(const char* str1, const char* str2, size_t n)
40{
41 size_t cmp = 0;
42 int ret = INT_MIN;
43
44 if (str1 == NULL || str2 == NULL)
45 {
46 return ret;
47 }
48
49 while ((*str1 || *str2) && cmp < n)
50 {
51 if ((ret = tolower((int)(*str1)) - tolower((int)(*str2))) != 0)
52 {
53 break;
54 }
55 ++cmp;
56 ++str1;
57 ++str2;
58 }
59
60 return ret;
61}
