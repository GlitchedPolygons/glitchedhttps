42 int glitchedhttps_strncmpic(const char* str1, const char* str2, size_t n)
43 {
44  size_t cmp = 0;
45  int ret = INT_MIN;
46 
47  if (str1 == NULL || str2 == NULL)
48  {
49  return ret;

52  while ((*str1 || *str2) && cmp < n)
53  {
54  if ((ret = tolower((int)(*str1)) - tolower((int)(*str2))) != 0)

57  }
