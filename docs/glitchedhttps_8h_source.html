59 static glitchedhttps_response* _glitchedhttps_parse_response_string(const chillbuff* response_string)
60 {
61  if (response_string == NULL)
62  {
63  return NULL;
64  }
65 
66  /* Allocate the output http response struct and set pointers to default value "NULL".
67  * The consumer of this returned value should not forget to call on this! */
68  glitchedhttps_response* out = malloc(sizeof(glitchedhttps_response));
69  if (out == NULL)
70  {
71  _glitchedhttps_log_error("OUT OF MEMORY!", __func__);
72  return NULL;
73  }
74 
75  out->raw = NULL;
76  out->date = NULL;
77  out->server = NULL;
78  out->headers = NULL;
79  out->content = NULL;
80  out->content_type = NULL;
81  out->content_encoding = NULL;
82  out->content_length = 0;
83  out->headers_count = 0;
84  out->status_code = -1;
85 
86  /* First of all, copy the whole, raw response string into the output. */
87  out->raw = malloc((response_string->length + 1) * response_string->element_size);
88  if (out->raw == NULL)
89  {
90  glitchedhttps_response_free(out);
91  _glitchedhttps_log_error("OUT OF MEMORY!", __func__);
92  return NULL;
93  }
94 
95  memcpy(out->raw, response_string->array, response_string->length);
96  out->raw[response_string->length] = '\0';
97 
98  /* Next comes the tedious parsing. */
99  const char delimiter[] = "\r\n";
100  const size_t delimiter_length = strlen(delimiter);
101 
102  const char content_delimiter[] = "\r\n\r\n";
103  const size_t content_delimiter_length = strlen(content_delimiter);
104 
105  char* current = response_string->array;
106  char* next = strstr(current, delimiter);
107 
108  bool parsed_status = false, parsed_server = false, parsed_date = false, parsed_content_type = false, parsed_content_encoding = false, parsed_content_length = false;
109 
110  chillbuff header_builder;
111  chillbuff_init(&header_builder, 16, sizeof(glitchedhttps_header), CHILLBUFF_GROW_DUPLICATIVE);
112 
113  while (next != NULL)
114  {
115  const size_t current_length = next - current;
116 
117  if (!parsed_status && glitchedhttps_strncmpic(current, "HTTP/", 5) == 0)
118  {
119  char n[4];
120  const char* c = memchr(current, ' ', current_length);
121  if (c != NULL)
122  {
123  n[3] = '\0';
124  memcpy(n, c + 1, 3);
125  out->status_code = atoi(n);
126  }
127  parsed_status = true;
128  }
129  else if (!parsed_server && glitchedhttps_strncmpic(current, "Server: ", 8) == 0)
130  {
131  const size_t out_length = current_length - 8;
132  out->server = malloc((out_length + 1) * sizeof(char));
133  if (out->server == NULL)
134  {
135  goto out_of_mem;
136  }
137  memcpy(out->server, current + 8, out_length);
138  out->server[out_length] = '\0';
139  chillbuff_push_back(&header_builder, glitchedhttps_header_init(current, 6, out->server, out_length), 1);
140  parsed_server = true;
141  }
142  else if (!parsed_date && glitchedhttps_strncmpic(current, "Date: ", 6) == 0)
143  {
144  const size_t out_length = current_length - 6;
145  out->date = malloc((out_length + 1) * sizeof(char));
146  if (out->date == NULL)
147  {
148  goto out_of_mem;
149  }
150  memcpy(out->date, current + 6, out_length);
151  out->date[out_length] = '\0';
152  chillbuff_push_back(&header_builder, glitchedhttps_header_init(current, 4, out->date, out_length), 1);
153  parsed_date = true;
154  }
155  else if (!parsed_content_type && glitchedhttps_strncmpic(current, "Content-Type: ", 14) == 0)
156  {
157  const size_t out_length = current_length - 14;
158  out->content_type = malloc((out_length + 1) * sizeof(char));
159  if (out->content_type == NULL)
160  {
161  goto out_of_mem;
162  }
163  memcpy(out->content_type, current + 14, out_length);
164  out->content_type[out_length] = '\0';
165  chillbuff_push_back(&header_builder, glitchedhttps_header_init(current, 12, out->content_type, out_length), 1);
166  parsed_content_type = true;
167  }
168  else if (!parsed_content_encoding && glitchedhttps_strncmpic(current, "Content-Encoding: ", 18) == 0)
169  {
170  const size_t out_length = current_length - 18;
171  out->content_encoding = malloc((out_length + 1) * sizeof(char));
172  if (out->content_encoding == NULL)
173  {
174  goto out_of_mem;
175  }
176  memcpy(out->content_encoding, current + 18, out_length);
177  out->content_encoding[out_length] = '\0';
178  chillbuff_push_back(&header_builder, glitchedhttps_header_init(current, 16, out->content_encoding, out_length), 1);
179  parsed_content_encoding = true;
180  }
181  else if (!parsed_content_length && glitchedhttps_strncmpic(current, "Content-Length: ", 16) == 0)
182  {
183  char n[64];
184  memset(n, '\0', sizeof(n));
185  const char* c = memchr(current, ' ', current_length);
186  if (c != NULL)
187  {
188  memcpy(n, c + 1, current_length - 1);
189  out->content_length = atoi(n);
190  snprintf(n, sizeof(n), "%zu", out->content_length);
191  chillbuff_push_back(&header_builder, glitchedhttps_header_init(current, 14, n, strlen(n)), 1);
192  }
193  parsed_content_length = true;
194  }
195  else if (current != response_string->array && strncmp(current - delimiter_length, content_delimiter, content_delimiter_length) == 0)
196  {
197  const char* content = (current - delimiter_length) + content_delimiter_length;
198  const size_t content_length = strlen(content);
199  out->content = malloc(content_length + 1);
200  if (out->content == NULL)
201  {
202  goto out_of_mem;
203  }
204  memcpy(out->content, content, content_length);
205  out->content[content_length] = '\0';
206  break; // If the content (request body) was found, it's time to stop. Because there won't be anything else to come.
207  }
208  else
209  {
210  const char header_separator[] = ": ";
211  const size_t header_separator_length = strlen(header_separator);
212 
213  const char* header_value = strstr(current, header_separator);
214  if (header_value != NULL)
215  {
216  const size_t header_type_length = header_value - current;
217  const size_t header_value_length = current_length - header_type_length - header_separator_length;
218  chillbuff_push_back(&header_builder, glitchedhttps_header_init(current, header_type_length, header_value + header_separator_length, header_value_length), 1);
219  }
220  }
221  current = next + delimiter_length;
222  next = strstr(current, delimiter);
223  }
224 
225  out->headers = malloc(sizeof(glitchedhttps_header) * header_builder.length);
226  if (out->headers == NULL)
227  {
228  goto out_of_mem;
229  }
230 
231  /* Copy the response headers into the output instance. */
232  out->headers_count = header_builder.length;
233  for (size_t i = 0; i < header_builder.length; i++)
234  {
235  const glitchedhttps_header h = ((glitchedhttps_header*)header_builder.array)[i];
236  out->headers[i] = *glitchedhttps_header_init(h.type, strlen(h.type), h.value, strlen(h.value));
237  }
238 
239  chillbuff_free(&header_builder);
240 
241  return out;
242 
243 out_of_mem:
244  _glitchedhttps_log_error("OUT OF MEMORY!", __func__);
245  glitchedhttps_response_free(out);
246  chillbuff_free(&header_builder);
247  return NULL;
248 }

251 static glitchedhttps_response* _glitchedhttps_https_request(const char* server_name, const int server_port, const char* request, const size_t buffer_size, const bool ssl_verification_optional)
252 {
253  if (server_name == NULL || request == NULL || server_port <= 0)
254  {
255  _glitchedhttps_log_error("INVALID HTTPS parameters passed into \"_glitchedhttps_https_request\". Returning NULL...", __func__);
256  return NULL;
257  }
258 
259  glitchedhttps_response* out = NULL;
260 
261  chillbuff response_string;
262  chillbuff_init(&response_string, 1024, sizeof(char), CHILLBUFF_GROW_DUPLICATIVE);
263 
264  uint32_t flags;
265  int ret = 1, length;
266  int exit_code = MBEDTLS_EXIT_FAILURE;
267  unsigned char buffer[buffer_size];
268 
269  time_t t;
270  srand((unsigned)time(&t));
271 
272  glitchedhttps_guid guid = glitchedhttps_new_guid(rand() & 1, rand() & 1);
273 
274  mbedtls_x509_crt cacert;
275  mbedtls_ssl_config ssl_config;
276  mbedtls_ssl_context ssl_context;
277  mbedtls_net_context net_context;
278  mbedtls_entropy_context entropy;
279  mbedtls_ctr_drbg_context ctr_drbg;
280 
281  mbedtls_net_init(&net_context);
282  mbedtls_ssl_init(&ssl_context);
283  mbedtls_ssl_config_init(&ssl_config);
284  mbedtls_x509_crt_init(&cacert);
285  mbedtls_ctr_drbg_init(&ctr_drbg);
286  mbedtls_entropy_init(&entropy);
287 
288  /* Seed the random number generator. */
289 
290  ret = mbedtls_ctr_drbg_seed(&ctr_drbg, mbedtls_entropy_func, &entropy, (const unsigned char*)guid.string, strlen(guid.string));
291  if (ret != 0)
292  {
293  char msg[128];
294  sprintf(msg, "HTTPS request failed: \"mbedtls_ctr_drbg_seed\" returned %d", ret);
295  _glitchedhttps_log_error(msg, __func__);
296  goto exit;
297  }
298 
299  /* Load the CA root certificates. */
300 
301  ret = mbedtls_x509_crt_parse(&cacert, (const unsigned char*)GLITCHEDHTTPS_CA_CERTS, sizeof(GLITCHEDHTTPS_CA_CERTS));
302  if (ret < 0)
303  {
304  char msg[128];
305  sprintf(msg, "HTTPS request failed: \"mbedtls_x509_crt_parse\" returned -0x%x", -ret);
306  _glitchedhttps_log_error(msg, __func__);
307  goto exit;
308  }
309 
310  /* Open the connection to the specified host. */
311 
312  char port[8];
313  memset(port, '\0', sizeof(port));
314  snprintf(port, sizeof(port), "%d", server_port);
315 
316  ret = mbedtls_net_connect(&net_context, server_name, port, MBEDTLS_NET_PROTO_TCP);
317  if (ret != 0)
318  {
319  char msg[128];
320  sprintf(msg, "HTTPS request failed: \"mbedtls_net_connect\" returned %d", ret);
321  _glitchedhttps_log_error(msg, __func__);
322  goto exit;
323  }
324 
325  /* Set up the SSL/TLS structure. */
326 
327  ret = mbedtls_ssl_config_defaults(&ssl_config, MBEDTLS_SSL_IS_CLIENT, MBEDTLS_SSL_TRANSPORT_STREAM, MBEDTLS_SSL_PRESET_DEFAULT);
328  if (ret != 0)
329  {
330  char msg[128];
331  sprintf(msg, "HTTPS request failed: \"mbedtls_ssl_config_defaults\" returned %d", ret);
332  _glitchedhttps_log_error(msg, __func__);
333  goto exit;
334  }
335 
336  mbedtls_ssl_conf_authmode(&ssl_config, ssl_verification_optional ? MBEDTLS_SSL_VERIFY_OPTIONAL : MBEDTLS_SSL_VERIFY_REQUIRED);
337  mbedtls_ssl_conf_ca_chain(&ssl_config, &cacert, NULL);
338  mbedtls_ssl_conf_rng(&ssl_config, mbedtls_ctr_drbg_random, &ctr_drbg);
339  mbedtls_ssl_conf_dbg(&ssl_config, &_glitchedhttps_debug, stdout);
340 
341  ret = mbedtls_ssl_setup(&ssl_context, &ssl_config);
342  if (ret != 0)
343  {
344  char msg[128];
345  sprintf(msg, "HTTPS request failed: \"mbedtls_ssl_setup\" returned %d", ret);
346  _glitchedhttps_log_error(msg, __func__);
347  goto exit;
348  }
349 
350  ret = mbedtls_ssl_set_hostname(&ssl_context, server_name);
351  if (ret != 0)
352  {
353  char msg[128];
354  sprintf(msg, "HTTPS request failed: \"mbedtls_ssl_set_hostname\" returned %d", ret);
355  _glitchedhttps_log_error(msg, __func__);
356  goto exit;
357  }
358 
359  mbedtls_ssl_set_bio(&ssl_context, &net_context, mbedtls_net_send, mbedtls_net_recv, NULL);
360 
361  /* SSL Handshake. */
362 
363  while ((ret = mbedtls_ssl_handshake(&ssl_context)) != 0)
364  {
365  if (ret != MBEDTLS_ERR_SSL_WANT_READ && ret != MBEDTLS_ERR_SSL_WANT_WRITE)
366  {
367  char msg[128];
368  sprintf(msg, "HTTPS request failed: \"mbedtls_ssl_handshake\" returned -0x%x", -ret);
369  _glitchedhttps_log_error(msg, __func__);
370  goto exit;
371  }
372  }
373 
374  /* Verify the server's X.509 certificate. */
375 
376  flags = mbedtls_ssl_get_verify_result(&ssl_context);
377  if (flags != 0)
378  {
379  char verification_buffer[1024];
380  mbedtls_x509_crt_verify_info(verification_buffer, sizeof(verification_buffer), " ! ", flags);
381  _glitchedhttps_log_error(verification_buffer, __func__);
382  goto exit;
383  }
384 
385  /* Write the request string.*/
386 
387  length = snprintf((char*)buffer, sizeof(buffer), "%s", request);
388 
389  while ((ret = mbedtls_ssl_write(&ssl_context, buffer, length)) <= 0)
390  {
391  if (ret != MBEDTLS_ERR_SSL_WANT_READ && ret != MBEDTLS_ERR_SSL_WANT_WRITE)
392  {
393  char msg[128];
394  sprintf(msg, "HTTPS request failed: \"mbedtls_ssl_write\" returned %d", ret);
395  _glitchedhttps_log_error(msg, __func__);
396  goto exit;
397  }
398  }
399 
400  /* Read the HTTP response. */
401 
402  for (;;)
403  {
404  length = (int)(sizeof(buffer) - 1);
405  memset(buffer, '\0', sizeof(buffer));
406  ret = mbedtls_ssl_read(&ssl_context, buffer, length);
407 
408  if (ret == MBEDTLS_ERR_SSL_WANT_READ || ret == MBEDTLS_ERR_SSL_WANT_WRITE)
409  {
410  continue;
411  }
412 
413  if (ret == MBEDTLS_ERR_SSL_PEER_CLOSE_NOTIFY)
414  {
415  break;
416  }
417 
418  if (ret < 0)
419  {
420  char msg[128];
421  sprintf(msg, "HTTPS request failed: \"mbedtls_ssl_read\" returned %d", ret);
422  _glitchedhttps_log_error(msg, __func__);
423  break;
424  }
425 
426  if (ret == 0)
427  {
428  /* EOF; ready to close the connection. */
429  break;
430  }
431 
432  length = ret;
433  chillbuff_push_back(&response_string, buffer, length);
434  }
435 
436  if (response_string.length == 0)
437  {
438  _glitchedhttps_log_error("HTTP response string empty!", __func__);
439  goto exit;
440  }
441 
442  out = _glitchedhttps_parse_response_string(&response_string);
443 
444  mbedtls_ssl_close_notify(&ssl_context);
445  exit_code = MBEDTLS_EXIT_SUCCESS;
446 
447 exit:
448 
449 #ifdef MBEDTLS_ERROR_C
450  if (exit_code != MBEDTLS_EXIT_SUCCESS)
451  {
452  char error_buf[256];
453  mbedtls_strerror(ret, error_buf, sizeof(error_buf));
454  char msg[1024];
455  sprintf(msg, "HTTPS request unsuccessful! Last error was: %d - %s", ret, error_buf);
456  _glitchedhttps_log_error(msg, __func__);
457  }
458 #endif
459 
460  mbedtls_net_free(&net_context);
461  mbedtls_x509_crt_free(&cacert);
462  mbedtls_ssl_free(&ssl_context);
463  mbedtls_ssl_config_free(&ssl_config);
464  mbedtls_ctr_drbg_free(&ctr_drbg);
465  mbedtls_entropy_free(&entropy);
466  chillbuff_free(&response_string);
467 
468  return (out);
469 }

570 glitchedhttps_response* glitchedhttps_submit(const glitchedhttps_request* request)
571 {
572  if (request->url == NULL)
573  {
574  _glitchedhttps_log_error("URL parameter NULL!", __func__);
575  return NULL;
576  }
577 
578  const bool https = glitchedhttps_is_https(request->url);
579  const char* server_host_ptr = https ? request->url + 8 : glitchedhttps_is_http(request->url) ? request->url + 7 : NULL;
580 
581  if (server_host_ptr == NULL)
582  {
583  _glitchedhttps_log_error("Missing or invalid protocol in passed URL: needs \"http://\" or \"https://\"", __func__);
584  return NULL;
585  }
586 
587  char server_host[256];
588  memset(server_host, '\0', sizeof(server_host));
589 
590  char* path = strchr(server_host_ptr, '/');
591  strncpy(server_host, server_host_ptr, path == NULL ? strlen(server_host_ptr) : path - server_host_ptr);
592 
593  int server_port = https ? 443 : 80;
594 
595  char* custom_port = strrchr(server_host, ':');
596  if (custom_port != NULL)
597  {
598  /* IPv6 safety check. */
599  if (*server_host_ptr != '[' || *(custom_port - 1) == ']')
600  {
601  server_port = atoi(custom_port + 1);
602  if (server_port <= 0 || server_port >= 65536)
603  {
604  char msg[128];
605  sprintf(msg, "Invalid port number \"%d\"", server_port);
606  _glitchedhttps_log_error(msg, __func__);
607  return NULL;
608  }
609  memset(custom_port, '\0', strlen(custom_port));
610  }
611  else
612  {
613  custom_port = NULL;
614  }
615  }
616 
617  if (path == NULL)
618  path = "/";
619 
620  char method[8];
621  if (!glitchedhttps_method_to_string(request->method, method, sizeof(method)))
622  {
623  _glitchedhttps_log_error("HTTP request submission rejected due to invalid HTTP method name.", __func__);
624  return NULL;
625  }
626  method[sizeof(method) - 1] = '\0';
627 
628  chillbuff request_string;
629  chillbuff_init(&request_string, 1024, sizeof(char), CHILLBUFF_GROW_DUPLICATIVE);
630 
631  char tmp[64 + strlen(method) + strlen(path) + strlen(server_host)];
632  snprintf(tmp, sizeof(tmp), "%s %s HTTP/1.1\r\nHost: %s\r\nConnection: Close\r\n", method, path, server_host);
633  chillbuff_push_back(&request_string, tmp, strlen(tmp));
634  memset(tmp, '\0', sizeof(tmp));
635 
636  for (size_t i = 0; i < request->additional_headers_count; i++)
637  {
638  glitchedhttps_header header = request->additional_headers[i];
639  char header_string[16 + strlen(header.type) + strlen(header.value)];
640  snprintf(header_string, sizeof(header_string), "%s: %s\r\n", header.type, header.value);
641  chillbuff_push_back(&request_string, header_string, strlen(header_string));
642  memset(header_string, '\0', sizeof(header_string));
