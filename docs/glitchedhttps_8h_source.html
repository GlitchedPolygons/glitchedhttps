131 int _glitchedhttps_parse_response_string(const chillbuff* response_string, glitchedhttps_response** out)
132 {
133  if (response_string == NULL)
134  {
135  _glitchedhttps_log_error("HTTP response parse error: \"response_string\" argument NULL; nothing to parse!", __func__);
136  return GLITCHEDHTTPS_RESPONSE_PARSE_ERROR;
137  }
138 
139  /* Allocate the output http response struct and set pointers to default value "NULL".
140  * The consumer of this returned value should not forget to call on this! */
141  glitchedhttps_response* response = malloc(sizeof(glitchedhttps_response));
142  if (response == NULL)
143  {
144  _glitchedhttps_log_error("OUT OF MEMORY!", __func__);
145  return GLITCHEDHTTPS_OUT_OF_MEM;
146  }
147 
148  response->raw = NULL;
149  response->date = NULL;
150  response->server = NULL;
151  response->headers = NULL;
152  response->content = NULL;
153  response->content_type = NULL;
154  response->content_encoding = NULL;
155  response->content_length = 0;
156  response->headers_count = 0;
157  response->status_code = -1;
158 
159  response->raw = malloc((response_string->length + 1) * response_string->element_size);
160  if (response->raw == NULL)
161  {
162  glitchedhttps_response_free(response);
163  _glitchedhttps_log_error("OUT OF MEMORY!", __func__);
164  return GLITCHEDHTTPS_OUT_OF_MEM;
165  }
166 
167  /* First of all, copy the whole, raw response string into the output. */
168  memcpy(response->raw, response_string->array, response_string->length);
169  response->raw[response_string->length] = '\0';
170 
171  /* Next comes the tedious parsing. */
172  const char header_delimiter[] = "\r\n";
173  const size_t header_delimiter_length = strlen(header_delimiter);
174 
175  const char content_delimiter[] = "\r\n\r\n";
176  const size_t content_delimiter_length = strlen(content_delimiter);
177 
178  char* current = response_string->array;
179  char* next = strstr(current, header_delimiter);
180 
181  bool parsed_status = false, parsed_server = false, parsed_date = false, parsed_content_type = false, parsed_content_encoding = false, parsed_content_length = false;
182 
183  chillbuff header_builder;
184  if (chillbuff_init(&header_builder, 16, sizeof(glitchedhttps_header), CHILLBUFF_GROW_DUPLICATIVE) != CHILLBUFF_SUCCESS)
185  {
186  _glitchedhttps_log_error("Chillbuff init failed: can't proceed without a proper request string builder... Perhaps go check out the chillbuff error logs!", __func__);
187  return GLITCHEDHTTPS_CHILLBUFF_ERROR;
188  }
189 
190  while (next != NULL)
191  {
192  const size_t current_length = next - current;
193 
194  if (!parsed_status && glitchedhttps_strncmpic(current, "HTTP/", 5) == 0)
195  {
196  char n[4];
197  const char* c = memchr(current, ' ', current_length);
198  if (c != NULL)
199  {
200  n[3] = '\0';
201  memcpy(n, c + 1, 3);
202  response->status_code = atoi(n);
203  }
204  parsed_status = true;
205  }
206  else if (!parsed_server && glitchedhttps_strncmpic(current, "Server: ", 8) == 0)
207  {
208  const size_t out_length = current_length - 8;
209  response->server = malloc((out_length + 1) * sizeof(char));
210  if (response->server == NULL)
211  {
212  goto out_of_mem;
213  }
214  memcpy(response->server, current + 8, out_length);
215  response->server[out_length] = '\0';
216  chillbuff_push_back(&header_builder, glitchedhttps_header_init(current, 6, response->server, out_length), 1);
217  parsed_server = true;
218  }
219  else if (!parsed_date && glitchedhttps_strncmpic(current, "Date: ", 6) == 0)
220  {
221  const size_t out_length = current_length - 6;
222  response->date = malloc((out_length + 1) * sizeof(char));
223  if (response->date == NULL)
224  {
225  goto out_of_mem;
226  }
227  memcpy(response->date, current + 6, out_length);
228  response->date[out_length] = '\0';
229  chillbuff_push_back(&header_builder, glitchedhttps_header_init(current, 4, response->date, out_length), 1);
230  parsed_date = true;
231  }
232  else if (!parsed_content_type && glitchedhttps_strncmpic(current, "Content-Type: ", 14) == 0)
233  {
234  const size_t out_length = current_length - 14;
235  response->content_type = malloc((out_length + 1) * sizeof(char));
236  if (response->content_type == NULL)
237  {
238  goto out_of_mem;
239  }
240  memcpy(response->content_type, current + 14, out_length);
241  response->content_type[out_length] = '\0';
242  chillbuff_push_back(&header_builder, glitchedhttps_header_init(current, 12, response->content_type, out_length), 1);
243  parsed_content_type = true;
244  }
245  else if (!parsed_content_encoding && glitchedhttps_strncmpic(current, "Content-Encoding: ", 18) == 0)
246  {
247  const size_t out_length = current_length - 18;
248  response->content_encoding = malloc((out_length + 1) * sizeof(char));
249  if (response->content_encoding == NULL)
250  {
251  goto out_of_mem;
252  }
253  memcpy(response->content_encoding, current + 18, out_length);
254  response->content_encoding[out_length] = '\0';
255  chillbuff_push_back(&header_builder, glitchedhttps_header_init(current, 16, response->content_encoding, out_length), 1);
256  parsed_content_encoding = true;
257  }
258  else if (!parsed_content_length && glitchedhttps_strncmpic(current, "Content-Length: ", 16) == 0)
259  {
260  char n[64];
261  memset(n, '\0', sizeof(n));
262  const char* c = memchr(current, ' ', current_length);
263  if (c != NULL)
264  {
265  memcpy(n, c + 1, current_length - 1);
266  response->content_length = atoi(n);
267  snprintf(n, sizeof(n), "%zu", response->content_length);
268  chillbuff_push_back(&header_builder, glitchedhttps_header_init(current, 14, n, strlen(n)), 1);
269  }
270  parsed_content_length = true;
271  }
272  else if (current != response_string->array && strncmp(current - header_delimiter_length, content_delimiter, content_delimiter_length) == 0)
273  {
274  const char* content = (current - header_delimiter_length) + content_delimiter_length;
275  const size_t content_length = strlen(content);
276  response->content = malloc(content_length + 1);
277  if (response->content == NULL)
278  {
279  goto out_of_mem;
280  }
281  memcpy(response->content, content, content_length);
282  response->content[content_length] = '\0';
283  break; // If the content (request body) was found, it's time to stop. Because there won't be anything else to come.
284  }
285  else
286  {
287  const char header_separator[] = ": ";
288  const size_t header_separator_length = strlen(header_separator);
289 
290  const char* header_value = strstr(current, header_separator);
291  if (header_value != NULL)
292  {
293  const size_t header_type_length = header_value - current;
294  const size_t header_value_length = current_length - header_type_length - header_separator_length;
295  chillbuff_push_back(&header_builder, glitchedhttps_header_init(current, header_type_length, header_value + header_separator_length, header_value_length), 1);
296  }
297  }
298  current = next + header_delimiter_length;
299  next = strstr(current, header_delimiter);
300  }
301 
302  response->headers = malloc(sizeof(glitchedhttps_header) * header_builder.length);
303  if (response->headers == NULL)
304  {
305  goto out_of_mem;
306  }
307 
308  /* Copy the response headers into the output instance. */
309  response->headers_count = header_builder.length;
310  for (size_t i = 0; i < header_builder.length; i++)
311  {
312  const glitchedhttps_header h = ((glitchedhttps_header*)header_builder.array)[i];
313  response->headers[i] = *glitchedhttps_header_init(h.type, strlen(h.type), h.value, strlen(h.value));
314  }
315 
316  *out = response;
317  chillbuff_free(&header_builder);
318  return GLITCHEDHTTPS_SUCCESS;
319 
320 out_of_mem:
321  _glitchedhttps_log_error("OUT OF MEMORY!", __func__);
322  glitchedhttps_response_free(response);
323  chillbuff_free(&header_builder);
324  return GLITCHEDHTTPS_OUT_OF_MEM;
325 }

328 int _glitchedhttps_https_request(const char* server_name, const int server_port, const char* request, const size_t buffer_size, const bool ssl_verification_optional, glitchedhttps_response** out)
329 {
330  if (server_name == NULL || request == NULL || server_port <= 0)
331  {
332  _glitchedhttps_log_error("INVALID HTTPS parameters passed into \"_glitchedhttps_https_request\". Returning NULL...", __func__);
333  return GLITCHEDHTTPS_INVALID_ARG;
334  }
335 
336  chillbuff response_string;
337  if (chillbuff_init(&response_string, 1024, sizeof(char), CHILLBUFF_GROW_DUPLICATIVE) != CHILLBUFF_SUCCESS)
338  {
339  _glitchedhttps_log_error("Chillbuff init failed: can't proceed without a proper request string builder... Perhaps go check out the chillbuff error logs!", __func__);
340  return GLITCHEDHTTPS_CHILLBUFF_ERROR;
341  }
342 
343  uint32_t flags;
344  int ret = 1, length, exit_code = -1;
345  int mbedtls_exit_code = MBEDTLS_EXIT_FAILURE;
346  unsigned char buffer[buffer_size];
347 
348  time_t t;
349  srand((unsigned)time(&t));
350 
351  glitchedhttps_guid guid = glitchedhttps_new_guid(rand() & 1, rand() & 1);
352 
353  mbedtls_x509_crt cacert;
354  mbedtls_ssl_config ssl_config;
355  mbedtls_ssl_context ssl_context;
356  mbedtls_net_context net_context;
357  mbedtls_entropy_context entropy;
358  mbedtls_ctr_drbg_context ctr_drbg;
359 
360  mbedtls_net_init(&net_context);
361  mbedtls_ssl_init(&ssl_context);
362  mbedtls_ssl_config_init(&ssl_config);
363  mbedtls_x509_crt_init(&cacert);
364  mbedtls_ctr_drbg_init(&ctr_drbg);
365  mbedtls_entropy_init(&entropy);
366 
367  /* Seed the random number generator. */
368 
369  ret = mbedtls_ctr_drbg_seed(&ctr_drbg, mbedtls_entropy_func, &entropy, (const unsigned char*)guid.string, strlen(guid.string));
370  if (ret != 0)
371  {
372  char msg[128];
373  sprintf(msg, "HTTPS request failed: \"mbedtls_ctr_drbg_seed\" returned %d", ret);
374  _glitchedhttps_log_error(msg, __func__);
375  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
376  goto exit;
377  }
378 
379  /* Load the CA root certificates. */
380 
381  ret = mbedtls_x509_crt_parse(&cacert, (const unsigned char*)GLITCHEDHTTPS_CA_CERTS, sizeof(GLITCHEDHTTPS_CA_CERTS));
382  if (ret < 0)
383  {
384  char msg[128];
385  sprintf(msg, "HTTPS request failed: \"mbedtls_x509_crt_parse\" returned -0x%x", -ret);
386  _glitchedhttps_log_error(msg, __func__);
387  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
388  goto exit;
389  }
390 
391  /* Open the connection to the specified host. */
392 
393  char port[8];
394  memset(port, '\0', sizeof(port));
395  snprintf(port, sizeof(port), "%d", server_port);
396 
397  ret = mbedtls_net_connect(&net_context, server_name, port, MBEDTLS_NET_PROTO_TCP);
398  if (ret != 0)
399  {
400  char msg[128];
401  sprintf(msg, "HTTPS request failed: \"mbedtls_net_connect\" returned %d", ret);
402  _glitchedhttps_log_error(msg, __func__);
403  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
404  goto exit;
405  }
406 
407  /* Set up the SSL/TLS structure. */
408 
409  ret = mbedtls_ssl_config_defaults(&ssl_config, MBEDTLS_SSL_IS_CLIENT, MBEDTLS_SSL_TRANSPORT_STREAM, MBEDTLS_SSL_PRESET_DEFAULT);
410  if (ret != 0)
411  {
412  char msg[128];
413  sprintf(msg, "HTTPS request failed: \"mbedtls_ssl_config_defaults\" returned %d", ret);
414  _glitchedhttps_log_error(msg, __func__);
415  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
416  goto exit;
417  }
418 
419  mbedtls_ssl_conf_authmode(&ssl_config, ssl_verification_optional ? MBEDTLS_SSL_VERIFY_OPTIONAL : MBEDTLS_SSL_VERIFY_REQUIRED);
420  mbedtls_ssl_conf_ca_chain(&ssl_config, &cacert, NULL);
421  mbedtls_ssl_conf_rng(&ssl_config, mbedtls_ctr_drbg_random, &ctr_drbg);
422  mbedtls_ssl_conf_dbg(&ssl_config, &_glitchedhttps_debug, stdout);
423 
424  ret = mbedtls_ssl_setup(&ssl_context, &ssl_config);
425  if (ret != 0)
426  {
427  char msg[128];
428  sprintf(msg, "HTTPS request failed: \"mbedtls_ssl_setup\" returned %d", ret);
429  _glitchedhttps_log_error(msg, __func__);
430  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
431  goto exit;
432  }
433 
434  ret = mbedtls_ssl_set_hostname(&ssl_context, server_name);
435  if (ret != 0)
436  {
437  char msg[128];
438  sprintf(msg, "HTTPS request failed: \"mbedtls_ssl_set_hostname\" returned %d", ret);
439  _glitchedhttps_log_error(msg, __func__);
440  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
441  goto exit;
442  }
443 
444  mbedtls_ssl_set_bio(&ssl_context, &net_context, mbedtls_net_send, mbedtls_net_recv, NULL);
445 
446  /* SSL Handshake. */
447 
448  while ((ret = mbedtls_ssl_handshake(&ssl_context)) != 0)
449  {
450  if (ret != MBEDTLS_ERR_SSL_WANT_READ && ret != MBEDTLS_ERR_SSL_WANT_WRITE)
451  {
452  char msg[128];
453  sprintf(msg, "HTTPS request failed: \"mbedtls_ssl_handshake\" returned -0x%x", -ret);
454  _glitchedhttps_log_error(msg, __func__);
455  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
456  goto exit;
457  }
458  }
459 
460  /* Verify the server's X.509 certificate. */
461 
462  flags = mbedtls_ssl_get_verify_result(&ssl_context);
463  if (flags != 0)
464  {
465  char verification_buffer[1024];
466  mbedtls_x509_crt_verify_info(verification_buffer, sizeof(verification_buffer), " ! ", flags);
467  _glitchedhttps_log_error(verification_buffer, __func__);
468  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
469  goto exit;
470  }
471 
472  /* Write the request string.*/
473 
474  length = snprintf((char*)buffer, sizeof(buffer), "%s", request);
475 
476  while ((ret = mbedtls_ssl_write(&ssl_context, buffer, length)) <= 0)
477  {
478  if (ret != MBEDTLS_ERR_SSL_WANT_READ && ret != MBEDTLS_ERR_SSL_WANT_WRITE)
479  {
480  char msg[128];
481  sprintf(msg, "HTTPS request failed: \"mbedtls_ssl_write\" returned %d", ret);
482  _glitchedhttps_log_error(msg, __func__);
483  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
484  goto exit;
485  }
486  }
487 
488  /* Read the HTTP response. */
489 
490  for (;;)
491  {
492  length = (int)(sizeof(buffer) - 1);
493  memset(buffer, '\0', sizeof(buffer));
494  ret = mbedtls_ssl_read(&ssl_context, buffer, length);
495 
496  if (ret == MBEDTLS_ERR_SSL_WANT_READ || ret == MBEDTLS_ERR_SSL_WANT_WRITE)
497  {
498  continue;
499  }
500 
501  if (ret == MBEDTLS_ERR_SSL_PEER_CLOSE_NOTIFY)
502  {
503  break;
504  }
505 
506  if (ret < 0)
507  {
508  char msg[128];
509  sprintf(msg, "HTTPS request failed: \"mbedtls_ssl_read\" returned %d", ret);
510  _glitchedhttps_log_error(msg, __func__);
511  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
512  break;
513  }
514 
515  if (ret == 0)
516  {
517  /* EOF; ready to close the connection. */
518  break;
519  }
520 
521  length = ret;
522  chillbuff_push_back(&response_string, buffer, length);
523  }
524 
525  if (response_string.length == 0)
526  {
527  _glitchedhttps_log_error("HTTP response string empty!", __func__);
528  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
529  goto exit;
530  }
531 
532  exit_code = _glitchedhttps_parse_response_string(&response_string, out);
533  mbedtls_ssl_close_notify(&ssl_context);
534  mbedtls_exit_code = MBEDTLS_EXIT_SUCCESS;
535 
536 exit:
537 
538 #ifdef MBEDTLS_ERROR_C
539  if (mbedtls_exit_code != MBEDTLS_EXIT_SUCCESS)
540  {
541  char error_buf[4096];
542  mbedtls_strerror(ret, error_buf, sizeof(error_buf));
543  char msg[8192];
544  snprintf(msg, sizeof(msg), "HTTPS request unsuccessful! Last error was: %d - %s", ret, error_buf);
545  _glitchedhttps_log_error(msg, __func__);
546  }
547 #endif
548 
549  mbedtls_net_free(&net_context);
550  mbedtls_x509_crt_free(&cacert);
551  mbedtls_ssl_free(&ssl_context);
552  mbedtls_ssl_config_free(&ssl_config);
553  mbedtls_ctr_drbg_free(&ctr_drbg);
554  mbedtls_entropy_free(&entropy);
555  chillbuff_free(&response_string);
556 
557  return exit_code;
558 }

561 int _glitchedhttps_http_request(const char* server_name, const int server_port, const char* request, const size_t buffer_size, glitchedhttps_response** out)
562 {
563  int exit_code;
564 
565  if (server_name == NULL || request == NULL || server_port <= 0)
566  {
567  _glitchedhttps_log_error("INVALID HTTP parameters passed into \"_glitchedhttps_http_request\".", __func__);
568  return GLITCHEDHTTPS_INVALID_ARG;
569  }
570 
571  chillbuff response_string;
572  if (chillbuff_init(&response_string, 1024, sizeof(char), CHILLBUFF_GROW_DUPLICATIVE) != CHILLBUFF_SUCCESS)
573  {
574  _glitchedhttps_log_error("Chillbuff init failed: can't proceed without a proper request string builder... Perhaps go check out the chillbuff error logs!", __func__);
575  return GLITCHEDHTTPS_CHILLBUFF_ERROR;
576  }
577 
578  if (response_string.length == 0)
579  {
580  _glitchedhttps_log_error("HTTP response string empty!", __func__);
581  exit_code = GLITCHEDHTTPS_INVALID_ARG;
582  goto exit;
583  }
584 
585 #if defined WIN32
586  WSADATA wsaData;
587  int iResult = WSAStartup(MAKEWORD(2, 2), &wsaData);
588  if (iResult != 0)
589  {
590  _glitchedhttps_log_error("Erros at \"WSAStartup\".", __func__);
591  return GLITCHEDHTTPS_EXTERNAL_ERROR;
592  }
593 #endif
594 
595  int csocket = socket(PF_INET, SOCK_STREAM, IPPROTO_TCP);
596  if (csocket < 0)
597  {
598  _glitchedhttps_log_error("Socket creation failed.", __func__);
599  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
600  goto exit;
601  }
602 
603  // Construct server address
604  struct sockaddr_in socket_address;
605  memset(&socket_address, 0, sizeof(socket_address));
606 
607  struct hostent* he = gethostbyname(server_name);
608  socket_address.sin_family = AF_INET;
609 
610  socket_address.sin_addr.s_addr = inet_addr("192.168.0.100"); // IP del server
611  socket_address.sin_port = htons(server_port);
612  // Connect to server.
613  if (connect(csocket, (struct sockaddr*)&socket_address, sizeof(socket_address)) < 0)
614  {
615  _glitchedhttps_log_error("Failed to connect to server.", __func__);
616  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
617  goto exit;
618  }
619 
620  char* inputString = "REPLACE ME ASAP";
621  size_t stringLen = strlen(inputString);
622 
623  if (send(csocket, inputString, stringLen, 0) != stringLen)
624  {
625  _glitchedhttps_log_error("send() sent a different number of bytes than expected", __func__);
626  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
627  goto exit;
628  }
629 
630  int bytesRcvd;
631  int totalBytesRcvd = 0;
632  char buf[512]; // buffer for data from the server
633  printf("Received: "); // Setup to print the echoed string
634 
635  while (totalBytesRcvd < stringLen)
636  {
637  if ((bytesRcvd = recv(csocket, buf, BUFFERSIZE - 1, 0)) <= 0)
638  {
639  _glitchedhttps_log_error("recv() failed or connection closed prematurely",__func__);
640  exit_code = GLITCHEDHTTPS_EXTERNAL_ERROR;
641  goto exit;
642  }
643  totalBytesRcvd += bytesRcvd; // Keep tally of total bytes
644  buf[bytesRcvd] = '\0'; // Add \0 so printf knows where to stop
645  printf("%s", buf); // Print the echo buffer
646  }
