582 static glitched_http_response* _glitched_http_parse_response_string(const chillbuff* response_string)
583 {
584  if (response_string == NULL)
585  {
586  return NULL;
587  }
588 
589  /* Allocate the output http response struct and set pointers to default value "NULL".
590  * The consumer of this returned value should not forget to call on this! */
591  glitched_http_response* out = malloc(sizeof(glitched_http_response));
592  if (out == NULL)
593  {
594  _glitched_http_log_error("OUT OF MEMORY!", __func__);
595  return NULL;
596  }
597 
598  out->raw = NULL;
599  out->date = NULL;
600  out->server = NULL;
601  out->headers = NULL;
602  out->content = NULL;
603  out->content_type = NULL;
604  out->content_encoding = NULL;
605  out->content_length = 0;
606  out->headers_count = 0;
607  out->status_code = -1;
608 
609  /* First of all, copy the whole, raw response string into the output. */
610  out->raw = malloc((response_string->length + 1) * response_string->element_size);
611  if (out->raw == NULL)
612  {
613  glitched_http_response_free(out);
614  _glitched_http_log_error("OUT OF MEMORY!", __func__);
615  return NULL;
616  }
617 
618  memcpy(out->raw, response_string->array, response_string->length);
619  out->raw[response_string->length] = '\0';
620 
621  /* Next comes the tedious parsing. */
622  const char delimiter[] = "\r\n";
623  const size_t delimiter_length = strlen(delimiter);
624 
625  const char content_delimiter[] = "\r\n\r\n";
626  const size_t content_delimiter_length = strlen(content_delimiter);
627 
628  char* current = response_string->array;
629  char* next = strstr(current, delimiter);
630 
631  bool parsed_status = false, parsed_server = false, parsed_date = false, parsed_content_type = false, parsed_content_encoding = false, parsed_content_length = false;
632 
633  chillbuff header_builder;
634  chillbuff_init(&header_builder, 16, sizeof(glitched_http_header), CHILLBUFF_GROW_DUPLICATIVE);
635 
636  while (next != NULL)
637  {
638  const size_t current_length = next - current;
639 
640  if (!parsed_status && glitched_http_strncmpic(current, "HTTP/", 5) == 0)
641  {
642  char n[4];
643  const char* c = memchr(current, ' ', current_length);
644  if (c != NULL)
645  {
646  n[3] = '\0';
647  memcpy(n, c + 1, 3);
648  out->status_code = atoi(n);
649  }
650  parsed_status = true;
651  }
652  else if (!parsed_server && glitched_http_strncmpic(current, "Server: ", 8) == 0)
653  {
654  const size_t out_length = current_length - 8;
655  out->server = malloc((out_length + 1) * sizeof(char));
656  if (out->server == NULL)
657  {
658  _glitched_http_log_error("OUT OF MEMORY!", __func__);
659  glitched_http_response_free(out);
660  chillbuff_free(&header_builder);
661  return NULL;
662  }
663  memcpy(out->server, current + 8, out_length);
664  out->server[out_length] = '\0';
665  chillbuff_push_back(&header_builder, glitched_http_header_init(current, 6, out->server, out_length), sizeof(glitched_http_header));
666  parsed_server = true;
667  }
668  else if (!parsed_date && glitched_http_strncmpic(current, "Date: ", 6) == 0)
669  {
670  const size_t out_length = current_length - 6;
671  out->date = malloc((out_length + 1) * sizeof(char));
672  if (out->date == NULL)
673  {
674  _glitched_http_log_error("OUT OF MEMORY!", __func__);
675  glitched_http_response_free(out);
676  chillbuff_free(&header_builder);
677  return NULL;
678  }
679  memcpy(out->date, current + 6, out_length);
680  out->date[out_length] = '\0';
681  chillbuff_push_back(&header_builder, glitched_http_header_init(current, 4, out->date, out_length), sizeof(glitched_http_header));
682  parsed_date = true;
683  }
684  else if (!parsed_content_type && glitched_http_strncmpic(current, "Content-Type: ", 14) == 0)
685  {
686  const size_t out_length = current_length - 14;
687  out->content_type = malloc((out_length + 1) * sizeof(char));
688  if (out->content_type == NULL)
689  {
690  _glitched_http_log_error("OUT OF MEMORY!", __func__);
691  glitched_http_response_free(out);
692  chillbuff_free(&header_builder);
693  return NULL;
694  }
695  memcpy(out->content_type, current + 14, out_length);
696  out->content_type[out_length] = '\0';
697  chillbuff_push_back(&header_builder, glitched_http_header_init(current, 12, out->content_type, out_length), sizeof(glitched_http_header));
698  parsed_content_type = true;
699  }
700  else if (!parsed_content_encoding && glitched_http_strncmpic(current, "Content-Encoding: ", 18) == 0)
701  {
702  const size_t out_length = current_length - 18;
703  out->content_encoding = malloc((out_length + 1) * sizeof(char));
704  if (out->content_encoding == NULL)
705  {
706  _glitched_http_log_error("OUT OF MEMORY!", __func__);
707  glitched_http_response_free(out);
708  chillbuff_free(&header_builder);
709  return NULL;
710  }
711  memcpy(out->content_encoding, current + 18, out_length);
712  out->content_encoding[out_length] = '\0';
713  chillbuff_push_back(&header_builder, glitched_http_header_init(current, 16, out->content_encoding, out_length), sizeof(glitched_http_header));
714  parsed_content_encoding = true;
715  }
716  else if (!parsed_content_length && glitched_http_strncmpic(current, "Content-Length: ", 16) == 0)
717  {
718  char n[64];
719  memset(n, '\0', sizeof(n));
720  const char* c = memchr(current, ' ', current_length);
721  if (c != NULL)
722  {
723  memcpy(n, c + 1, current_length - 1);
724  out->content_length = atoi(n);
725  snprintf(n, sizeof(n), "%lld", out->content_length);
726  chillbuff_push_back(&header_builder, glitched_http_header_init(current, 14, n, strlen(n)), sizeof(glitched_http_header));
727  }
728  parsed_content_length = true;
729  }
730  else if (current != response_string->array && strncmp(current - delimiter_length, content_delimiter, content_delimiter_length) == 0)
731  {
732  const char* content = (current - delimiter_length) + content_delimiter_length;
733  const size_t content_length = strlen(content);
734  out->content = malloc(content_length + 1);
735  if (out->content == NULL)
736  {
737  _glitched_http_log_error("OUT OF MEMORY!", __func__);
738  glitched_http_response_free(out);
739  chillbuff_free(&header_builder);
740  return NULL;
741  }
742  memcpy(out->content, content, content_length);
743  out->content[content_length] = '\0';
744  break; // If the content (request body) was found, it's time to stop. Because there won't be anything else to come.
745  }
746  else
747  {
748  const char header_separator[] = ": ";
749  const size_t header_separator_length = strlen(header_separator);
750 
751  const char* header_value = strstr(current, header_separator);
752  if (header_value != NULL)
753  {
754  const size_t header_type_length = header_value - current;
755  const size_t header_value_length = current_length - header_type_length - header_separator_length;
756  chillbuff_push_back(&header_builder, glitched_http_header_init(current, header_type_length, header_value + header_separator_length, header_value_length), sizeof(glitched_http_header));
757  }
758  }
759  current = next + delimiter_length;
760  next = strstr(current, delimiter);
761  }
762 
763  out->headers = malloc(sizeof(glitched_http_header) * header_builder.length); // TODO: copy headers array kinda here?!?!
764  if (out->headers == NULL)
765  {
766  _glitched_http_log_error("OUT OF MEMORY!", __func__);
767  glitched_http_response_free(out);
768  chillbuff_free(&header_builder);
769  return NULL;
770  }
771 
772  chillbuff_free(&header_builder);
773 
774  return out;
775 }
