170 static inline int chillbuff_init(chillbuff* buff, const size_t initial_capacity, const size_t element_size, const chillbuff_growth_method growth_method)
171 {
172  if (buff == NULL)
173  {
174  _chillbuff_printerr("Tried to init a NULL chillbuff instance; wouldn't end well. Cancelled...", __func__);
175  return CHILLBUFF_NULL_ARG;
176  }
177 
178  if (element_size == 0)
179  {
180  _chillbuff_printerr("Storing elements of size \"0\" makes no sense...", __func__);
181  return CHILLBUFF_INVALID_ARG;
182  }
183 
184  if (growth_method < 0 || growth_method > 3)
185  {
186  _chillbuff_printerr("Invalid grow method! Please use the appropriate chillbuff_growth_method enum!", __func__);
187  return CHILLBUFF_INVALID_ARG;
188  }
189 
190  buff->length = 0;
191  buff->element_size = element_size;
192  buff->growth_method = growth_method;
193  buff->capacity = initial_capacity == 0 ? 16 : initial_capacity;
194  buff->array = calloc(buff->capacity, buff->element_size);
195 
196  if (buff->array == NULL)
197  {
198  _chillbuff_printerr("OUT OF MEMORY!", __func__);
199  return CHILLBUFF_OUT_OF_MEM;
200  }
201 
202  return CHILLBUFF_SUCCESS;
203 }

247 static int chillbuff_push_back(chillbuff* buff, const void* elements, const size_t elements_count)
248 {
249  if (buff == NULL)
250  {
251  _chillbuff_printerr("Tried to append to a NULL chillbuff instance!", __func__);
252  return CHILLBUFF_NULL_ARG;
253  }
254 
255  if (elements == NULL)
256  {
257  _chillbuff_printerr("Tried to append NULL element(s) to a chillbuff instance!", __func__);
258  return CHILLBUFF_NULL_ARG;
259  }
260 
261  if (elements_count == 0)
262  {
263  _chillbuff_printerr("The passed \"elements_count\" argument is zero; nothing to append!", __func__);
264  return CHILLBUFF_INVALID_ARG;
265  }
266 
267  for (size_t i = 0; i < elements_count; i++)
268  {
269  if (buff->length == buff->capacity)
270  {
271  size_t new_capacity;
272 
273  switch (buff->growth_method)
274  {
275  default:
276  _chillbuff_printerr("Invalid grow method! Please use the appropriate chillbuff_growth_method enum!", __func__);
277  return CHILLBUFF_INVALID_ARG;
278  case CHILLBUFF_GROW_DUPLICATIVE:
279  new_capacity = (buff->capacity * 2);
280  break;
281  case CHILLBUFF_GROW_TRIPLICATIVE:
282  new_capacity = (buff->capacity * 3);
283  break;
284  case CHILLBUFF_GROW_LINEAR:
285  new_capacity = (buff->capacity + buff->element_size);
286  break;
287  case CHILLBUFF_GROW_EXPONENTIAL:
288  new_capacity = (buff->capacity * buff->capacity);
289  break;
290  }
291 
292  if (new_capacity <= buff->capacity || new_capacity >= UINT64_MAX / buff->element_size)
293  {
294  _chillbuff_printerr("Couldn't push back due to buffer OVERFLOW!", __func__);
295  return CHILLBUFF_OVERFLOW;
296  }
297 
298  void* new_array = realloc(buff->array, new_capacity * buff->element_size);
299  if (new_array == NULL)
300  {
301  _chillbuff_printerr("Couldn't resize chillbuff underlying array; OUT OF MEMORY!", __func__);
302  return CHILLBUFF_OUT_OF_MEM;
303  }
304 
305  memset(new_array + (buff->element_size * buff->length), '\0', (new_capacity - buff->length) * buff->element_size);
306  buff->array = new_array;
307  buff->capacity = new_capacity;
308  }
309 
310  memcpy(buff->array + (buff->element_size * buff->length++), elements + (i * buff->element_size), buff->element_size);
311  }
312 
313  return CHILLBUFF_SUCCESS;
314 }
